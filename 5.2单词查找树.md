# 单词查找树
* 由字符串键中的所有字符构造而成,允许使用被查找键中的字符进行查找
* 每条链接都对应着一个字符,将每个键(*单词作为键*)所关联的值保存在该键**最后一个字母**所对应的结点中
* 值为空的结点在符号表中没有对应的键,他们的存在是为了简化单词查找树中的查找操作

```java
package com.company.string;

import edu.princeton.cs.algs4.Queue;

public class TrieST<Value> {
    private static int R = 256;
    private Node root;

    /**
     * R向单词查找树
     * 结点不保存字符值,而是根据结点在R向数组中的位置来确定所代表的字符值
     */
    private static class Node {
        private Object val;
        private Node[] next = new Node[R];
    }

    /**
     * 返回单词查找树键值对的数量
     */
    private int size(Node x) {
        if (x == null) {
            return 0;
        }

        int cnt = 0;
        if (x.val != null) {
            cnt++;
        }

        for (char c = 0; c < R; c++) {
            cnt += size(x.next[c]);
        }

        return cnt;
    }

    public int size() {
        return size(root);
    }

    /**
     * 返回以x作为根节点的子单词查找树中与key相关联的尾结点
     *
     * @param x
     * @param key
     * @param d   目标字符串的开始查找的下标(第d个字符,从0计数),每次递归调用都+1
     * @return
     */
    private Node get(Node x, String key, int d) {
        if (x == null)
            return null;

        //执行到这里时说明x一路都不为null
        //这时如果d == key.length(),说明查找到了目标的最后一个字符,返回这个结点
        if (d == key.length())
            return x;

        //找到第d个字符所对应的子单词查找树
        //先获取目标String key中第d个字符
        //然后获取这个char字符的ASCII数值作为该结点的R向数组中的下标
        char c = key.charAt(d);
        return get(x.next[c], key, d + 1);
    }

    public Value get(String key) {
        Node x = get(root, key, 0);
        if (x == null)
            return null;
        return (Value) x.val;
    }

    /**
     * put新单词,如果key存在于以x为根节点的子单词查找树中则更新它对应的val
     *
     * @param x
     * @param key
     * @param val
     * @param d   目标字符串的开始查找的下标(第d个字符,从0计数)
     * @return
     */
    private Node put(Node x, String key, Value val, int d) {
        if (x == null)
            x = new Node();

        //这里找到了(或者是new的一个新的)尾结点,更新尾结点的val
        if (d == key.length()) {
            x.val = val;
            return x;
        }

        //找到第d个字符所对应的子单词查找树
        char c = key.charAt(d);
        x.next[c] = put(x.next[c], key, val, d + 1);
        return x;
    }

    public void put(String key, Value val) {
        root = put(root, key, val, 0);
    }

    /**
     * 从单词查找树中删除一个键值对
     *
     * @param x
     * @param key
     * @param d
     * @return
     */
    private Node delete(Node x, String key, int d) {
        if (x == null)
            return null;

        //先找到键所对应的尾结点并将它的值置空
        if (d == key.length()) {
            x.val = null;
        } else {
            char c = key.charAt(d);
            x.next[c] = delete(x.next[c], key, d + 1);
        }

        //如果该结点的值和所有的链接均为空则返回null,否则返回x
        //这里考虑的是遍历到目标尾结点开始回归时,
        //在回归的路上遇到的其他的尾结点
        if (x.val != null)
            return x;
        for (char c = 0; c < R; c++) {
            if (x.next[c] != null)
                return x;
        }
        return null;
    }

    public void delete(String key) {
        root = delete(root, key, 0);
    }

    /**
     * 维护一个字符串用来保存从根节点出发的路径上的一系列字符,把每个单词保存在队列中
     *
     * @param x
     * @param pre 和该结点关联的字符串(从根节点到该结点的路径上的所有字符)
     * @param q
     */
    private void collect(Node x, String pre, Queue<String> q) {
        if (x == null)
            return;

        //如果找到了尾结点,把从根节点到这个尾结点的所有字符加入队列
        if (x.val != null)
            q.enqueue(pre);

        //对当前结点的所有字典开始遍历
        for (char c = 0; c < R; c++) {
            collect(x.next[c], pre + c, q);
        }
    }

    /**
     * 多了一个通配符参数的重载方法,用于通配符匹配方法keysThatMatch()
     *
     * @param x
     * @param pre
     * @param pat 指定匹配的模式
     * @param q
     */
    private void collect(Node x, String pre, String pat, Queue<String> q) {
        //d每次递归调用都会+1
        int d = pre.length();
        if (x == null)
            return;

        //如果前缀的长度等于pat的长度 且 当前节点是一个尾结点
        if (d == pat.length() && x.val != null)
            q.enqueue(pre);

        //不考虑长度超过模式字符串的键
        if (d == pat.length())
            return;

        //取出pat的第d个字符(去掉前面已经遍历过的前缀)
        char next = pat.charAt(d);
        for (char c = 0; c < R; c++) {
            //如果pat中含有通配符,就需要递归调用处理所有的链接
            if (next == '.' || next == c) {
                collect(x.next[c], pre + c, pat, q);
            }
        }
    }

    /**
     * 收集符号表中所有符合前缀目标的键
     *
     * @param pre 前缀目标
     * @return
     */
    public Iterable<String> keysWithPrefix(String pre) {
        Queue<String> q = new Queue<>();
        //首先获取到根节点中通过查找pre得到的尾结点(通过这个尾结点来实现符合前缀)
        Node tail = get(root, pre, 0);
        //然后从这个尾结点出发,获取所有单词
        collect(tail, pre, q);
        return q;
    }

    /**
     * 收集一棵单词查找树中所有键的轨迹
     *
     * @return
     */
    public Iterable<String> keys() {
        //因为""的length是0,所以get方法得到的是根节点
        //然后从根节点出发查找所有的单词
        return keysWithPrefix("");
    }

    /**
     * 获取所有和s匹配的键,其中"."匹配任意字符
     *
     * @param pat
     * @return
     */
    public Iterable<String> keysThatMatch(String pat) {
        Queue<String> q = new Queue<>();
        collect(root, "", pat, q);
        return q;
    }

    /**
     * 找到给定字符串在单词查找树中的最长键前缀
     * 记录查找路径上所找到的最长键的长度
     * 将它作为递归方法的参数在遇到 值非空的结点时 更新它
     *
     * @param x
     * @param s
     * @param d      递归调用次数的计数器
     * @param length 最长键的长度
     */
    private int search(Node x, String s, int d, int length) {
        //在被查找的字符串结束或者遇到空链接不匹配时终止
        if (x == null)
            return length;
        //遇到尾结点时就更新最长键的长度
        if (x.val != null)
            length = d;
        if (d == s.length())
            return length;

        char c = s.charAt(d);
        return search(x.next[c], s, d + 1, length);
    }

    /**
     * s的前缀中最长的键
     *
     * @param s
     * @return
     */
    public String longestPrefixOf(String s) {
        int length = search(root, s, 0, 0);
        return s.substring(0, length);
    }
}
```
